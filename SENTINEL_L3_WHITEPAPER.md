function _forwardPropagation(int256[] memory input) internal view returns (int256[] memory) {
    int256[] memory currentInput = input;

    for (uint256 layerIdx = 0; layerIdx < neuralLayers.length; layerIdx++) {
        NeuralLayer memory layer = neuralLayers[layerIdx];
        int256[] memory layerOutput = new int256[](layer.neurons);

```solidity
function _calculateTrend(uint256[] memory data) internal pure returns (uint256) {
    if (data.length < 2) return 100; // Neutral trend

    uint256 firstHalf = _calculateMean(data[0:data.length/2]);
    uint256 secondHalf = _calculateMean(data[data.length/2:data.length]);

    if (secondHalf > firstHalf) {
        uint256 increase = ((secondHalf - firstHalf) * 100) / firstHalf;
        return 100 + Math.min(increase, 100);
    } else {
        uint256 decrease = ((firstHalf - secondHalf) * 100) / firstHalf;
        return Math.max(100 - decrease, 0);
    }
}

function _calculateVolatility(uint256[] memory data) internal pure returns (uint256) {
    uint256 mean = _calculateMean(data);
    uint256 variance = 0;

    for (uint256 i = 0; i < data.length; i++) {
        uint256 diff = data[i] > mean ? data[i] - mean : mean - data[i];
        variance += diff * diff;
    }

    variance /= data.length;
    uint256 stdDev = _sqrt(variance);

    return (stdDev * 100) / mean; // Coefficient of variation
}
```

        coveragePeriod: coveragePeriod,
        startTime: block.timestamp,
